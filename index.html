Número mayor utilizando métodos, se pueden ingresar los números manualmente ó generarlos aleatoreamente


package segundosemestre;
import java.util.Scanner;
import java.util.Random;
public class NumMayorMetodos {
public static void main(String[] args) {
Scanner sc = new Scanner(System.in);
int arreglo[][];
arreglo= new int [4][4];

randomOIngresar(arreglo);
System.out.println("--------");
desplegar(arreglo);
System.out.println("--------");
System.out.println("");
System.out.println("El número mayor es: "+ mayor(arreglo));
}

public static int[][] randomOIngresar(int arreglo[][]){
Scanner sc=new Scanner(System.in);
System.out.println("¿Qué desea?");
System.out.println("(1) Ingresar los números");
System.out.println("(2) Generar los números aleatoreamente");
System.out.println("");
int resp;
resp = sc.nextInt();
switch (resp){
case 1:
for (int i = 0; i < 4; i++) {
for (int j = 0; j < 4; j++) {
System.out.println("Igrese el número: ");
arreglo[i][j]=sc.nextInt();
}
}
break;
case 2:
for (int i = 0; i < 4; i++) {
for (int j = 0; j < 4; j++) {
int limite=1000;
Random rand= new Random();
arreglo[i][j]= rand.nextInt(limite);
}
}
break;
}
return(arreglo);
}

public static void desplegar(int gato[][]){
for (int i = 0; i < 4; i++) {
for (int j = 0; j < 4; j++) {
System.out.print(gato[i][j]+" ");
}
System.out.println("");
}
}

public static int mayor(int arreglo[][]){
int mayor= arreglo[0][0];
for (int i = 0; i < 4; i++) {
for (int j = 0; j < 4; j++) {
if (arreglo[i][j]> mayor) {
mayor = arreglo[i][j];
}
}
}
return(mayor);
}
}
